\label{codeblock:figure}\begin{minted}{jl}
function ICDF(u)
    u = 1.0 - u

    g = 1 + (4u) * (2u + sqrt(1 + 4u^2))
    n = exp2(log2(g) * (-1/3))
    p = g * n^2
    c = 1 + p + n
    sr = (3 / log2(MathConstants.e)) * log2(c / (4u))

    rcpExp = c^3 / ((4u) * (c^2 + 16 * u^2))

    return sr, 8pi * rcpExp
end
\end{minted}
